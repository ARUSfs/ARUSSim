125void Simulator::check_lap() {
126 double x = vehicle_dynamics_.x_;
127 double y = vehicle_dynamics_.y_;
128
129 // Calculate the signed distance from the vehicle to the line equation through the TPLs
130 double dist_to_tpl = y - tpl_coef_a_ * x - tpl_coef_b_;
131
132 // Calculate the distance from the vehicle to the midpoint between the TPLs
133 double d = std::sqrt(std::pow(x - mid_tpl_x_, 2) + std::pow(y - mid_tpl_y_, 2));
134
135 if (dist_to_tpl * prev_dist_to_tpl_ < 0 and d < 5) {
136 // Publish the result
137 std_msgs::msg::Bool msg;
138 msg.data = true;
139 lap_signal_pub_->publish(msg);
140 }
141 prev_dist_to_tpl_ = dist_to_tpl;
142}

166void Simulator::on_slow_timer()
167{
168 double x = vehicle_dynamics_.x_;
169 double y = vehicle_dynamics_.y_;
170 double yaw = vehicle_dynamics_.yaw_;
171
172 // Update track
173 sensor_msgs::msg::PointCloud2 track_msg;
174 pcl::toROSMsg(track_, track_msg);
175 track_msg.header.stamp = clock_->now();
176 track_msg.header.frame_id="arussim/world";
177 track_pub_->publish(track_msg);
178
179 fixed_trajectory_pub_->publish(fixed_trajectory_msg_);
180
181 // Random noise generation
182 std::random_device rd;
183 std::mt19937 gen(rd());
184 std::normal_distribution<> dist(0.0, kNoisePerception);
185
186 auto perception_cloud = pcl::PointCloud<ConeXYZColorScore>();
187 for (auto &point : track_.points)
188 {
189 double d = std::sqrt(std::pow(point.x - (x + kPosLidarX*std::cos(yaw)), 2) + std::pow(point.y - (y + kPosLidarX*std::sin(yaw)), 2));
190 if (d < kFOV)
191 {
192 ConeXYZColorScore p;
193 p.x = (point.x - x)*std::cos(yaw) + (point.y - y)*std::sin(yaw) + dist(gen) - kPosLidarX;
194 p.y = -(point.x - x)*std::sin(yaw) + (point.y - y)*std::cos(yaw) + dist(gen);
195 p.z = 0.0;
196 p.color = point.color;
197 p.score = 1.0;
198 if (p.x > kMinPerceptionX) {
199 perception_cloud.push_back(p);
200 }
201 if (p.x >= kCOGBackDist && p.x <= kCOGFrontDist && p.y >= -kCarWidth && p.y <= kCarWidth)
202 {
203 arussim_msgs::msg::PointXY msg;
204 msg.x = point.x;
205 msg.y = point.y;
206 hit_cones_pub_->publish(msg);
207 }
208 }
209 }
210
211 sensor_msgs::msg::PointCloud2 perception_msg;
212 pcl::toROSMsg(perception_cloud, perception_msg);
213 perception_msg.header.stamp = clock_->now();
214 perception_msg.header.frame_id="arussim/vehicle_cog";
215 perception_pub_->publish(perception_msg);

374{
375 // Create a TransformStamped message
376 geometry_msgs::msg::TransformStamped transform_stamped;
377
378 // Set the frame ID and child frame ID
379 transform_stamped.header.stamp = clock_->now();
380 transform_stamped.header.frame_id = "arussim/world";
381 transform_stamped.child_frame_id = "arussim/vehicle_cog";

384 transform_stamped.transform.translation.x = vehicle_dynamics_.x_;
385 transform_stamped.transform.translation.y = vehicle_dynamics_.y_;
386 transform_stamped.transform.translation.z = 0.0;
387

389 tf2::Quaternion q;
390 q.setRPY(0, 0, vehicle_dynamics_.yaw_); // Roll, Pitch, Yaw in radians
391 transform_stamped.transform.rotation.x = q.x();
392 transform_stamped.transform.rotation.y = q.y();
393 transform_stamped.transform.rotation.z = q.z();
394 transform_stamped.transform.rotation.w = q.w();
395
396 // Broadcast the transform
397 tf_broadcaster_->sendTransform(transform_stamped);

405void Simulator::load_track(const std_msgs::msg::String::SharedPtr track_msg)
406{
407 // Extract and remove the ".pcd" suffix if present
408 track_name_ = track_msg->data;
409 if(track_name_.size() >= 4 && track_name_.substr(track_name_.size()-4) == ".pcd"){
410 track_name_.resize(track_name_.size()-4);
411 }
412
413 // Load the track point cloud using the .pcd extension
414 std::string package_path = ament_index_cpp::get_package_share_directory("arussim");
415 std::string filename = package_path + "/resources/tracks/" + track_name_ + ".pcd";
416 if (pcl::io::loadPCDFile<ConeXYZColorScore>(filename, track_) == -1)
417 {
418 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "Couldn't read file %s", filename.c_str());
419 return;
420 }
421 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Loaded %ld data points from %s",
422 track_.points.size(), filename.c_str());
423
424 // Clear previous TPL cones
425 tpl_cones_.clear();
426
427 // Extract the TPLs
428 for (const auto& point : track_.points)
429 {
430 if (point.color == 4)
431 {
432 tpl_cones_.push_back(std::make_pair(point.x, point.y));
433 }
434 }
435 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "TPLs: %ld", tpl_cones_.size());
436
437 if (tpl_cones_.size() != 2) {
438 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "tpl_cones_ does not contain exactly 2 points.");
439 use_tpl_ = false;
440 } else {
441 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "TPL1: (%f, %f), TPL2: (%f, %f)",
442 tpl_cones_[0].first, tpl_cones_[0].second, tpl_cones_[1].first, tpl_cones_[1].second);
443 use_tpl_ = true;
444
445 // Coordinates of the two cones
446 double x1 = tpl_cones_[0].first;
447 double y1 = tpl_cones_[0].second;
448 double x2 = tpl_cones_[1].first;
449 double y2 = tpl_cones_[1].second;
450
451 // Calculate slope (a) and y-intercept (b)
452 tpl_coef_a_ = (y2 - y1) / (x2 - x1 + 0.000001); // Avoid division by zero
453 tpl_coef_b_ = y1 - tpl_coef_a_ * x1;
454
455 // Calculate midpoint between the two cones
456 mid_tpl_x_ = (x1 + x2) / 2.0;
457 mid_tpl_y_ = (y1 + y2) / 2.0;
458 }
459
460 // Extract the fixed trajectory using the .json extension
461 std::string json_filename = package_path + "/resources/tracks/" + track_name_ + ".json";
462 std::ifstream tray_json(json_filename);
463 if (!tray_json.is_open()) {
464 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "Couldn't open JSON trajectory file %s", json_filename.c_str());
465 return;
466 }
467
468 nlohmann::json tray_data;
469 tray_json >> tray_data;
470 tray_json.close();
471
472 std::vector<float> traj_x = tray_data["x"].get<std::vector<float>>();
473 std::vector<float> traj_y = tray_data["y"].get<std::vector<float>>();
474 std::vector<float> traj_s = tray_data["s"].get<std::vector<float>>();
475 std::vector<float> traj_k = tray_data["k"].get<std::vector<float>>();
476 std::vector<float> traj_speed_profile = tray_data["speed_profile"].get<std::vector<float>>();
477 std::vector<float> traj_acc_profile = tray_data["acc_profile"].get<std::vector<float>>();
478
479 if (traj_x.size() == traj_y.size() && traj_x.size() > 0) {
480 for (size_t i = 0; i < traj_x.size(); i++) {
481 arussim_msgs::msg::PointXY point;
482 point.x = traj_x[i];
483 point.y = traj_y[i];
484 fixed_trajectory_msg_.points.push_back(point);
485 }
486 }
487
488 if (traj_s.size() == traj_k.size() && traj_s.size() > 0) {
489 for (size_t i = 0; i < traj_s.size(); i++) {
490 fixed_trajectory_msg_.s.push_back(traj_s[i]);
491 fixed_trajectory_msg_.k.push_back(traj_k[i]);
492 }
493 }
494
495 if (traj_speed_profile.size() > 0) {
496 for (size_t i = 0; i < traj_speed_profile.size(); i++) {
497 fixed_trajectory_msg_.speed_profile.push_back(traj_speed_profile[i]);
498 }
499 }
500
501 if (traj_acc_profile.size() > 0) {
502 for (size_t i = 0; i < traj_acc_profile.size(); i++) {
503 fixed_trajectory_msg_.acc_profile.push_back(traj_acc_profile[i]);
504 }
505 }
506
507}
