44 def lineseg_dists(self, p, a, b):
45 """Cartesian distance from point to line segment
46
47 Edited to support arguments as series, from:
48 https://stackoverflow.com/a/54442561/11208892
49
50 Args:
51 - p: np.array of single point, shape (2,) or 2D array, shape (x, 2)
52 - a: np.array of shape (x, 2)
53 - b: np.array of shape (x, 2)
54 """
55 # normalized tangent vectors
56 d_ba = b - a
57 d = np.divide(d_ba, (np.hypot(d_ba[:, 0], d_ba[:, 1])
58 .reshape(-1, 1)))
59
60 # signed parallel distance components
61 # rowwise dot products of 2D vectors
62 s = np.multiply(a - p, d).sum(axis=1)
63 t = np.multiply(p - b, d).sum(axis=1)
64
65 # clamped parallel distance
66 h = np.maximum.reduce([s, t, np.zeros(len(s))])
67
68 # perpendicular distance component
69 # rowwise cross products of 2D vectors
70 d_pa = p - a
71 c = d_pa[:, 0] * d[:, 1] - d_pa[:, 1] * d[:, 0]
72
73 return np.hypot(h, c)

164 def getCurvature(self, point1, point2, point3):
165 # Calculating length of all three sides
166 len_side_1 = round( np.linalg.norm(point1-point2), 2)
167 len_side_2 = round( np.linalg.norm(point2-point3), 2)
168 len_side_3 = round( np.linalg.norm(point1-point3), 2)
169
170 # sp is semi-perimeter
171 sp = (len_side_1 + len_side_2 + len_side_3) / 2
172
173 # Calculating area using Herons formula
174 area = np.sqrt(max(sp * (sp - len_side_1) * (sp - len_side_2) * (sp - len_side_3),0.00001))
175
176 # Calculating curvature using Menger curvature formula
177 curvature = (4 * area) / max(len_side_1 * len_side_2 * len_side_3,0.0000001)
178
179 return curvature
