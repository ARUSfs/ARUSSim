198void PlotInterface::onInitialize()
199{
200 // Access the abstract ROS Node and
201 // in the process lock it for exclusive use until the method is done.
202 node_ptr_ = getDisplayContext()->getRosNodeAbstraction().lock();
203
204 // Get a pointer to the familiar rclcpp::Node for making subscriptions/publishers
205 // (as per normal rclcpp code)
206 rclcpp::Node::SharedPtr node = node_ptr_->get_raw_node();
207
208 // Subscribers
209 torque_sub_ = node->create_subscription<arussim_msgs::msg::FourWheelDrive>(
210 "/arussim/torque4WD", 1,
211 [this](const arussim_msgs::msg::FourWheelDrive::SharedPtr msg) {

213 update_telemetry_bar(msg->front_right, msg->front_left, msg->rear_right, msg->rear_left);

251bool PlotInterface::eventFilter(QObject* obj, QEvent* event)
252{
253 if(obj == gg_graph_label_) {
254 if (event->type() == QEvent::Wheel) {
255 auto* wheel_event = static_cast<QWheelEvent*>(event);
256 double angle_delta = wheel_event->angleDelta().y();
257 gg_zoom_factor_ *= (angle_delta > 0 ? 1.1 : 0.9);
258 gg_zoom_factor_ = std::clamp(gg_zoom_factor_, 0.01, 4.0);
259 } else if (event->type() == QEvent::MouseButtonPress) {
260 auto* mouse_event = static_cast<QMouseEvent*>(event);
261 if(mouse_event->button() == Qt::LeftButton) {
262 gg_last_mouse_pos_ = mouse_event->pos();
263 }
264 } else if (event->type() == QEvent::MouseMove && QEvent::MouseButtonPress) {
265 auto* mouse_event = static_cast<QMouseEvent*>(event);
266 // Calculate the position of (0,0) in the graph based on gg_center_x_, gg_center_y_, and gg_zoom_factor_
267 QPoint delta = mouse_event->pos() - gg_last_mouse_pos_;
268 gg_center_x_ -= delta.x() / (10.0 / gg_zoom_factor_);
269 gg_center_y_ += delta.y() / (10.0 / gg_zoom_factor_);
270 gg_last_mouse_pos_ = mouse_event->pos();
271 }
272 } else if (obj == speed_graph_label_) {
273 if (event->type() == QEvent::Wheel) {
274 auto* wheel_event = static_cast<QWheelEvent*>(event);
275 double angle_delta = wheel_event->angleDelta().y();
276 // Accumulate delta until sensibility param is reached
277 double accumulated_delta = 0.0;
278 accumulated_delta += angle_delta;
279 const double sensibility = 30.0;
280 if (accumulated_delta >= sensibility) {
281 max_vx_ += 5.0;
282 accumulated_delta = 0.0;
283 } else if (accumulated_delta <= -sensibility) {
284 max_vx_ -= 5.0;
285 accumulated_delta = 0.0;
286 }
287 }
288 }
289 return QObject::eventFilter(obj, event);
290}

399void PlotInterface::update_vx_target_graph()
400{
401 // Get the elapsed time in seconds from the start
402 double current_time = timer_.elapsed() / 1000.0;
403
404 // Add the new data point
405 vx_history_.append(qMakePair(current_time, vx_));
406 target_speed_history_.append(qMakePair(current_time, target_speed_));
407
408 // Remove points older than 10 seconds
409 while (!vx_history_.isEmpty() && (current_time - vx_history_.first().first > 10.0) && (current_time - target_speed_history_.first().first > 10.0))
410 {
411 vx_history_.removeFirst();
412 target_speed_history_.removeFirst();
413 }
414
415 // Configure plot dimensions
416 int pixmap_width = speed_graph_label_->width();
417 int pixmap_height = speed_graph_label_->height();
418 QPixmap pixmap(pixmap_width, pixmap_height);
419 pixmap.fill(Qt::white);
420
421 QPainter painter(&pixmap);
422 painter.setRenderHint(QPainter::Antialiasing);
423
424 // Draw grid lines
425 int num_rows = max_vx_ / 5.0;
426 double step_y = pixmap_height / static_cast<double>(num_rows);
427 painter.setPen(QPen(Qt::lightGray, graph_grid_width_, Qt::DashLine));
428 for (int j = 1; j < num_rows; ++j) {
429 painter.drawLine(0, j * step_y, pixmap_width, j * step_y);
430 }
431
432 // Draw numbers 5, 10, 15, 20 at the same height as each row of the grid
433 painter.setPen(Qt::black);
434 for (int j = 1; j <= num_rows; ++j) {
435 painter.drawText(0, pixmap_height - j * step_y, QString::number(j * 5));
436 }
437
438 if(vx_history_.isEmpty()){
439 speed_graph_label_->setPixmap(pixmap);
440 return;
441 }
442
443 // Draw vx line in red
444 painter.setPen(QPen(Qt::red, pen_size_));
445 QPainterPath vx_path;
446 bool first_vx_point = true;
447 for (const auto &point : vx_history_)
448 {
449 double x = ((point.first - (current_time - 10.0)) / 10.0) * pixmap_width;
450 double norm = (point.second - min_vx_) / (max_vx_ - min_vx_);
451 double y = pixmap_height - (norm * pixmap_height);
452 if(first_vx_point) {
453 vx_path.moveTo(x, y);
454 first_vx_point = false;
455 } else {
456 vx_path.lineTo(x, y);
457 }
458 }
459 painter.drawPath(vx_path);
460
461 // Draw target speed line in blue
462 painter.setPen(QPen(Qt::blue, pen_size_));
463 QPainterPath ts_path;
464 bool first_ts_point = true;
465 for (const auto &point : target_speed_history_)
466 {
467 double x = ((point.first - (current_time - 10.0)) / 10.0) * pixmap_width;
468 double norm = (point.second - min_vx_) / (max_vx_ - min_vx_);
469 double y = pixmap_height - (norm * pixmap_height);
470 if(first_ts_point) {
471 ts_path.moveTo(x, y);
472 first_ts_point = false;
473 } else {
474 ts_path.lineTo(x, y);
475 }
476 }
477 painter.drawPath(ts_path);
478
479 // Draw legend
480 QString legend_text_target = tr("Target");
481 QString legend_text_vx = tr("Vx");
482 int legend_width = std::max(QFontMetrics(painter.font()).horizontalAdvance(legend_text_target)*2,
483 QFontMetrics(painter.font()).horizontalAdvance(legend_text_vx)*2);
484 int legend_height = QFontMetrics(painter.font()).height()*2.5;
485 QRect legend_rect(20, 10, legend_width, legend_height);
486 painter.setPen(Qt::black);
487 painter.setBrush(QColor(255, 255, 255, 200));
488 painter.drawRect(legend_rect);
489
490 int line_length = legend_width / 4;
491 int y_target = legend_rect.top() + legend_height / 3;
492 int y_vx = legend_rect.top() + 2 * legend_height / 3;
493
494 // Legend font size
495 QFont legend_font = painter.font();
496 legend_font.setPointSizeF(std::min(pixmap_height * 0.075, 12.0));
497 painter.setFont(legend_font);
498
499 // Legend for Target Speed (blue)
500 painter.setPen(QPen(Qt::blue, pen_size_*0.9));
501 painter.drawLine(legend_rect.left() + 10, y_target, legend_rect.left() + 10 + line_length, y_target);
502 painter.setPen(Qt::black);
503 painter.drawText(legend_rect.left() + 10 + line_length + 10, y_target + 5, legend_text_target);
504
505 // Legend for Vx (red)
506 painter.setPen(QPen(Qt::red, pen_size_));
507 painter.drawLine(legend_rect.left() + 10, y_vx, legend_rect.left() + 10 + line_length, y_vx);
508 painter.setPen(Qt::black);
509 painter.drawText(legend_rect.left() + 10 + line_length + 10, y_vx + 5, legend_text_vx);
510
511 speed_graph_label_->setPixmap(pixmap);
512}

514void PlotInterface::update_gg_graph()
515{
516 // Remove points older than 30 seconds
517 if (vx_ != 0.5){
518 gg_vector_.append(std::make_tuple(ay_, ax_, vx_));
519 if (!timer_gg_started_){
520 timer_gg_.start();
521 timer_gg_started_ = true;
522 }
523 } if (timer_gg_.elapsed() > 30000) {
524 gg_vector_.removeFirst();
525 }
526
527 if (!gg_graph_label_) return;
528
529 int pixmap_width = gg_graph_label_->width();
530 int pixmap_height = gg_graph_label_->height();
531 QPixmap pixmap(pixmap_width, pixmap_height);
532 pixmap.fill(Qt::white);
533
534 QPainter painter(&pixmap);
535 painter.setRenderHint(QPainter::Antialiasing);
536
537 // Calculate the position of (0,0) in the graph based on gg_center_x_, gg_center_y_, and gg_zoom_factor_
538 double zero_x = ((0.0 - gg_center_x_) + 12.0 * gg_zoom_factor_) / (24.0 * gg_zoom_factor_) * pixmap_width;
539 double zero_y = pixmap_height - ((0.0 - gg_center_y_) + 12.0 * gg_zoom_factor_) / (24.0 * gg_zoom_factor_) * pixmap_height;
540
541 // Draw the grid applying translation and zoom
542 for (int i = -50; i <= 50; i++)
543 {
544 if(i == 0) continue;
545 double x = ((i - gg_center_x_) + 12.0 * gg_zoom_factor_) / (24.0 * gg_zoom_factor_) * pixmap_width;
546 double y = pixmap_height - ((i - gg_center_y_) + 12.0 * gg_zoom_factor_) / (24.0 * gg_zoom_factor_) * pixmap_height;
547 painter.setPen(QPen(Qt::lightGray, graph_grid_width_));
548 painter.drawLine(x, 0, x, pixmap_height);
549 painter.drawLine(0, y, pixmap_width, y);
550 }
551
552 // Draw axes centered at the real (0,0)
553 painter.setPen(QPen(Qt::black, graph_grid_width_*2));
554 painter.drawLine(0, zero_y, pixmap_width, zero_y);
555 painter.drawLine(zero_x, 0, zero_x, pixmap_height);
556
557 // draw GG legend
558 QString legend_text = tr("GG diagram");
559 int legend_width = QFontMetrics(painter.font()).horizontalAdvance(legend_text)*1.25;
560 int legend_height = QFontMetrics(painter.font()).height()*1.1;
561 QRect legend_gg_rect(10, 10, legend_width, legend_height);
562 painter.setPen(Qt::black);
563 painter.setBrush(QColor(255, 255, 255, 200));
564 painter.drawRect(legend_gg_rect);
565 QFont legend_font = painter.font();
566 legend_font.setPointSizeF(std::min(pixmap_height * 0.075, 10.0));
567 painter.setFont(legend_font);
568 painter.drawText(legend_gg_rect.adjusted(10, 0, 0, 0), legend_text);
569
570 // Draw points from gg_vector_ with new scaling for range -12 to 12
571 for (auto &p : gg_vector_) {
572 double loc_ay = std::get<0>(p);
573 double loc_ax = std::get<1>(p);
574 double loc_vx = std::get<2>(p);
575
576 // Normalize loc_vx using min_vx_ and max_vx_
577 double norm = (loc_vx - min_vx_) / (max_vx_ - min_vx_);
578 norm = std::clamp(norm, 0.0, 1.0);
579 // Map normalized value to hue: 240 (blue) to 0 (red)
580 int hue = static_cast<int>(240 * (1 - norm));
581 QColor point_color = QColor::fromHsv(hue, 255, 255);
582
583 painter.setPen(QPen(point_color, pen_size_));
584 double rx = ((loc_ay - gg_center_x_) + 12.0 * gg_zoom_factor_)
585 / (24.0 * gg_zoom_factor_) * pixmap_width;
586 double ry = pixmap_height -
587 ((loc_ax - gg_center_y_) + 12.0 * gg_zoom_factor_)
588 / (24.0 * gg_zoom_factor_) * pixmap_height;
589 painter.drawPoint(QPointF(rx, ry));
590 }
591
592 gg_graph_label_->setPixmap(pixmap);
593
594 // Draw horizontal multicolor bar legend below the graph
595 int legend_bar_height = 20;
596 int legend_bar_width = gg_legend_label_->width();
597 QPixmap legend_pixmap(legend_bar_width, legend_bar_height + 20);
598 legend_pixmap.fill(Qt::white);
599 QPainter legend_painter(&legend_pixmap);
600 legend_painter.setRenderHint(QPainter::Antialiasing);
601
602 for (int i = 0; i < legend_bar_width; ++i) {
603 double norm = static_cast<double>(i) / legend_bar_width;
604 int hue = static_cast<int>(240 * (1 - norm));
605 QColor color = QColor::fromHsv(hue, 255, 255);
606 legend_painter.setPen(QPen(color, 1));
607 legend_painter.drawLine(i, 0, i, legend_bar_height);
608 }
609
610 // Draw velocity values
611 for (double v = min_vx_; v <= max_vx_; v += 2.5) {
612 double norm = (v - min_vx_) / (max_vx_ - min_vx_);
613 int x = static_cast<int>(norm * legend_bar_width);
614 legend_painter.setPen(Qt::black);
615 legend_painter.drawText(x, legend_bar_height + 15, QString::number(v, 'f', 1));
616 }
617
618 gg_legend_label_->setPixmap(legend_pixmap);
619}
