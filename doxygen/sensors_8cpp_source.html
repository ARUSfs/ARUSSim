138void Sensors::torque_cmd_timer(){
139 // Random noise generation with different noise for each wheel speed
140 std::random_device rd;
141 std::mt19937 gen(rd());
142
143 // Corrected noise distributions
144 std::normal_distribution<> dist_fr(0.0, kNoiseTorqueFrontRight);
145 std::normal_distribution<> dist_fl(0.0, kNoiseTorqueFrontLeft);
146 std::normal_distribution<> dist_rr(0.0, kNoiseTorqueRearRight);
147 std::normal_distribution<> dist_rl(0.0, kNoiseTorqueRearLeft);
148
149 // Apply noise to the state variables
150 torque_cmd_.fr_ = torque_cmd.front_right + dist_fr(gen);
151 torque_cmd_.fl_ = torque_cmd.front_left + dist_fl(gen);
152 torque_cmd_.rr_ = torque_cmd.rear_right + dist_rr(gen);
153 torque_cmd_.rl_ = torque_cmd.rear_left + dist_rl(gen);
154
155 // Create the torque message
156 auto message = arussim_msgs::msg::FourWheelDrive();
157
158 message.front_right = torque_cmd_.fr_;
159 message.front_left = torque_cmd_.fl_;
160 message.rear_right = torque_cmd_.rr_;
161 message.rear_left = torque_cmd_.rl_;
162
163 // Publish the torque message
164 torque_pub_->publish(message);
165}

203void Sensors::wheel_speed_timer()
204{
205 // Random noise generation with different noise for each wheel speed
206 std::random_device rd;
207 std::mt19937 gen(rd());
208 std::normal_distribution<> dist_front_right(0.0, kNoiseWheelSpeedFrontRight);
209 std::normal_distribution<> dist_front_left(0.0, kNoiseWheelSpeedFrontLeft);
210 std::normal_distribution<> dist_rear_right(0.0, kNoiseWheelSpeedRearRight);
211 std::normal_distribution<> dist_rear_left(0.0, kNoiseWheelSpeedRearLeft);
212
213 // Apply noise to the state variables
214 wheel_speed_.fr_ = wheel_speed.front_right * 0.202 + dist_front_right(gen);
215 wheel_speed_.fl_ = wheel_speed.front_left * 0.202 + dist_front_left(gen);
216 wheel_speed_.rr_ = wheel_speed.rear_right * 0.202 + dist_rear_right(gen);
217 wheel_speed_.rl_ = wheel_speed.rear_left * 0.202 + dist_rear_left(gen);
