99def profiles(x_smooth, y_smooth, v_max=10.0, ax_max=5.0, ay_max=5.0):
100 '''
101 Calculate json data for the circuit
102 '''
103 acum = 0.0
104 s = [0.0]
105 xp, yp = [], []
106 for i in range(len(x_smooth) - 1):
107 dx = x_smooth[i+1] - x_smooth[i]
108 dy = y_smooth[i+1] - y_smooth[i]
109 dist = math.sqrt(dx*dx + dy*dy)
110 xp.append(dx)
111 yp.append(dy)
112 acum += dist
113 s.append(acum)
114 xp.append(xp[-1])
115 yp.append(yp[-1])
116
117 xpp, ypp = [], []
118 for i in range(len(xp) - 1):
119 xpp.append(xp[i+1] - xp[i])
120 ypp.append(yp[i+1] - yp[i])
121 xpp.append(xpp[-1])
122 ypp.append(ypp[-1])
123
124 k = []
125 for i in range(len(xpp)):
126 den = ((xp[i]**2 + yp[i]**2)**1.5)
127 val = 0 if den == 0 else (xp[i]*ypp[i] - xpp[i]*yp[i]) / den
128 k.append(val)
129
130 # Speed limits
131 speed_profile = [0.0 for _ in s]
132 v_grip = [min(math.sqrt(ay_max / max(abs(val), 1e-4)), v_max) for val in k]
133 speed_profile[0] = 1.0
134
135 for j in range(1, len(speed_profile)):
136 ds = s[j] - s[j-1]
137 speed_profile[j] = math.sqrt(speed_profile[j-1]**2 + 2*ax_max*ds)
138 if speed_profile[j] > v_grip[j]:
139 speed_profile[j] = v_grip[j]
140 for j in range(len(speed_profile)-2, -1, -1):
141 ds = s[j+1] - s[j]
142 v_brake = math.sqrt(speed_profile[j+1]**2 + 2*ax_max*ds)
143 if speed_profile[j] > v_brake:
144 speed_profile[j] = v_brake
145
146 acc_profile = [0.0]*len(speed_profile)
147 for i in range(len(speed_profile) - 1):
148 ds = s[i+1] - s[i]
149 if ds != 0:
150 acc_profile[i] = (speed_profile[i+1]**2 - speed_profile[i]**2) / (2 * ds)
151
152 trajectory_json_data = {
153 "x": list(x_smooth),
154 "y": list(y_smooth),
155 "s": s,
156 "k": k,
157 "speed_profile": speed_profile,
158 "acc_profile": acc_profile
159 }

161 return trajectory_json_data
