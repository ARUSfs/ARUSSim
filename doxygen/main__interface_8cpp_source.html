135void MainInterface::onInitialize()
136{
137 // Access the abstract ROS Node and
138 // in the process lock it for exclusive use until the method is done.
139 node_ptr_ = getDisplayContext()->getRosNodeAbstraction().lock();
140
141 // Get a pointer to the familiar rclcpp::Node for making subscriptions/publishers
142 // (as per normal rclcpp code)
143 rclcpp::Node::SharedPtr node = node_ptr_->get_raw_node();
144
145 // Publishers
146 reset_pub_ = node->create_publisher<std_msgs::msg::Bool>("/arussim/reset", 1);
147 circuit_pub_ = node->create_publisher<std_msgs::msg::String>("/arussim/circuit", 1);
148
149 // Subscribers
150 lap_time_sub_ = node->create_subscription<std_msgs::msg::Float32>(
151 "/arussim/lap_time", 1,
152 [this](const std_msgs::msg::Float32::SharedPtr msg) {
153 QMetaObject::invokeMethod(this, [this, msg]() {
154 update_lap_time_labels(msg->data);
155 }, Qt::QueuedConnection);
156 }
157 );
158
159 hit_cones_bool_sub_ = node->create_subscription<std_msgs::msg::Bool>(
160 "/arussim/hit_cones_bool", 1,
161 [this](const std_msgs::msg::Bool::SharedPtr msg) {
162 if (msg->data) {
163 QMetaObject::invokeMethod(this, [this]() {
164 hit_cones_counter_++;
165 hit_cones_label_->setText("Hit cones: " + QString::number(hit_cones_counter_));
166 }, Qt::QueuedConnection);
167 }
168 }
169 );
170}

177void MainInterface::update_lap_time_labels(double lap_time_)
178{
179 lap_counter_++;
180 lap_label_->setText("Lap: " + QString::number(lap_counter_));
181
182 if (lap_time_ < best_lap_time_) {
183 best_lap_time_ = lap_time_;
184 best_lt_label_->setText("Best Lap Time: " + QString::number(best_lap_time_, 'f', 3) + "s");
185 last_lt_label_->setStyleSheet("color: purple;");
186 } else if (lap_time_ < last_lap_time_) {
187 last_lap_time_ = lap_time_;
188 last_lt_label_->setStyleSheet("color: green;");
189 } else {
190 last_lt_label_->setStyleSheet("color: black;");
191 }
192 last_lt_label_->setText("Last Lap Time: " + QString::number(lap_time_, 'f', 3) + "s");
193}

200void MainInterface::launch_button_clicked()
201{
202 if (simulation_process_ == nullptr) {
203 simulation_process_ = new QProcess(this);
204 // Merge standard output and error
205 simulation_process_->setProcessChannelMode(QProcess::MergedChannels);
206 // Connect signal to capture output
207 connect(simulation_process_, &QProcess::readyReadStandardOutput, this, &MainInterface::process_output);
208
209 QString launch_file = launch_select_->currentText();
210 QStringList args;
211 args << "launch" << "common_meta" << launch_file;
212 simulation_process_->start("ros2", args);
213 }
214}
