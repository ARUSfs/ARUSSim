36void VehicleDynamics::calculate_dynamics(){
37
38 calculate_tire_slip();
39 calculate_tire_loads();
40
41 Tire_force force_fl = calculate_tire_forces(tire_slip_.alpha_fl_, tire_slip_.lambda_fl_, tire_loads_.fl_);
42 Tire_force force_fr = calculate_tire_forces(tire_slip_.alpha_fr_, tire_slip_.lambda_fr_, tire_loads_.fr_);
43 Tire_force force_rl = calculate_tire_forces(tire_slip_.alpha_rl_, tire_slip_.lambda_rl_, tire_loads_.rl_);
44 Tire_force force_rr = calculate_tire_forces(tire_slip_.alpha_rr_, tire_slip_.lambda_rr_, tire_loads_.rr_);
45
46 force.fy = force_fl.fy;
47 force.fx = force_fl.fx;
48
49 double fy_front = force_fl.fy + force_fr.fy;
50 double fy_rear = force_rl.fy + force_rr.fy;
51
52 ax_ = calculate_fx(force_fl, force_fr, force_rl, force_rr) / kMass;
53 double total_fy = fy_front * std::cos(delta_) + fy_rear;
54 total_fy += (force_fl.fx + force_fr.fx) * std::sin(delta_);
55 ay_ = total_fy / kMass;
56
57 x_dot_ = vx_ * std::cos(yaw_) - vy_ * std::sin(yaw_);
58 y_dot_ = vx_ * std::sin(yaw_) + vy_ * std::cos(yaw_);
59
60 vx_dot_ = ax_ + r_ * vy_;
61 vy_dot_ = ay_ - r_ * vx_;
62
63 double mz_lateral = fy_front * std::cos(delta_) * kLf - fy_rear * kLr;
64 mz_lateral += (force_fl.fy - force_fr.fy) * std::sin(delta_) * kTrackWidth/2;
65 double mz_longitudinal = ((force_fr.fx - force_fl.fx) * std::cos(delta_) + force_rr.fx - force_rl.fx) * kTrackWidth/2;
66 mz_longitudinal += (force_fl.fx + force_fr.fx) * std::sin(delta_) * kLf;
67
68 double total_mz = mz_lateral + mz_longitudinal;
69
70 r_dot_ = total_mz / kIzz;
71
72 // Tire angular acceleration
73 w_fl_dot_ = (torque_cmd_.fl_ - force_fl.fx * kTireDynRadius) / kTireInertia;
74 w_fr_dot_ = (torque_cmd_.fr_ - force_fr.fx * kTireDynRadius) / kTireInertia;
75 w_rl_dot_ = (torque_cmd_.rl_ - force_rl.fx * kTireDynRadius) / kTireInertia;
76 w_rr_dot_ = (torque_cmd_.rr_ - force_rr.fx * kTireDynRadius) / kTireInertia;
77
78 delta_dot_ = std::clamp(delta_v_, -kSteeringVMax, kSteeringVMax);
79 delta_v_dot_ = - kCoefDelta * delta_ - kCoefV * delta_v_ + kCoefInput * input_delta_;
80 delta_v_dot_ = std::clamp(delta_v_dot_, -kSteeringAMax, kSteeringAMax);

110

121void VehicleDynamics::calculate_tire_loads(){
122
123 double load_transfer_ay = kMass * kHCog * ay_ / kTrackWidth;
124 double load_transfer_ax = kMass * kHCog * ax_ / kWheelBase;
125
126 tire_loads_.fl_ = kStaticLoadFront - (1 - kMassDistributionRear) * load_transfer_ay - load_transfer_ax/2;
127 tire_loads_.fr_ = kStaticLoadFront + (1 - kMassDistributionRear) * load_transfer_ay - load_transfer_ax/2;
128 tire_loads_.rl_ = kStaticLoadFront - kMassDistributionRear * load_transfer_ay + load_transfer_ax/2;
129 tire_loads_.rr_ = kStaticLoadFront + kMassDistributionRear * load_transfer_ay + load_transfer_ax/2;
130
131 double aero_lift = 0.5 * kAirDensity * kCLA * vx_*vx_;
132 double aero_drag = 0.5 * kAirDensity * kCDA * vx_*vx_;
133
134 tire_loads_.fl_ += kCOPx * aero_lift / 2 - (kCOPy - kHCog) * aero_drag;
135 tire_loads_.fr_ += kCOPx * aero_lift / 2 - (kCOPy - kHCog) * aero_drag;
136 tire_loads_.rl_ += (1 - kCOPx) * aero_lift / 2 + (kCOPy - kHCog) * aero_drag;
137 tire_loads_.rr_ += (1 - kCOPx) * aero_lift / 2 + (kCOPy - kHCog) * aero_drag;
138
139
140 if(tire_loads_.fl_ < 0){tire_loads_.fl_ = 0;}
141 if(tire_loads_.fr_ < 0){tire_loads_.fr_ = 0;}
142 if(tire_loads_.rl_ < 0){tire_loads_.rl_ = 0;}
143 if(tire_loads_.rr_ < 0){tire_loads_.rr_ = 0;}

147void VehicleDynamics::calculate_tire_slip(){
148 double vy_front = vy_ + kLf * r_;
149 double vy_rear = vy_ - kLr * r_;
150 double vx_left = vx_ - r_ * kTrackWidth / 2;
151 double vx_right = vx_ + r_ * kTrackWidth / 2;
152
153 if(vx_ > 1){
154 tire_slip_.alpha_fl_ = std::atan(vy_front / vx_left) - delta_;
155 tire_slip_.alpha_fr_ = std::atan(vy_front / vx_right) - delta_;
156 tire_slip_.alpha_rl_ = std::atan(vy_rear / vx_left);
157 tire_slip_.alpha_rr_ = std::atan(vy_rear / vx_right);
158 }
159
160 double vx_fl = std::sqrt(vy_front*vy_front + vx_left*vx_left) * std::cos(tire_slip_.alpha_fl_);
161 double vx_fr = std::sqrt(vy_front*vy_front + vx_right*vx_right) * std::cos(tire_slip_.alpha_fr_);
162 double vx_rl = std::sqrt(vy_rear*vy_rear + vx_left*vx_left) * std::cos(tire_slip_.alpha_rl_);
163 double vx_rr = std::sqrt(vy_rear*vy_rear + vx_right*vx_right) * std::cos(tire_slip_.alpha_rr_);
164
165 double eps = 0.01;
166
167 tire_slip_.lambda_fl_ = kTireDynRadius * wheel_speed_.fl_ / (vx_fl + eps) - 1;
168 tire_slip_.lambda_fr_ = kTireDynRadius * wheel_speed_.fr_ / (vx_fr + eps) - 1;
169 tire_slip_.lambda_rl_ = kTireDynRadius * wheel_speed_.rl_ / (vx_rl + eps) - 1;
170 tire_slip_.lambda_rr_ = kTireDynRadius * wheel_speed_.rr_ / (vx_rr + eps) - 1;
171
172 if(vx_ < 0.1){
173 tire_slip_.lambda_fl_ = kTireDynRadius * wheel_speed_.fl_ / (vx_fl + eps);
174 tire_slip_.lambda_fr_ = kTireDynRadius * wheel_speed_.fr_ / (vx_fr + eps);
175 tire_slip_.lambda_rl_ = kTireDynRadius * wheel_speed_.rl_ / (vx_rl + eps);
176 tire_slip_.lambda_rr_ = kTireDynRadius * wheel_speed_.rr_ / (vx_rr + eps);
177 }
178
179}

181VehicleDynamics::Tire_force VehicleDynamics::calculate_tire_forces(double slip_angle, double slip_ratio, double tire_load){
182
183 // Combined slip
184 double alpha_star = slip_angle / pac_param_.kAlphaP;
185 double lambda_star = slip_ratio / pac_param_.kLambdaP;
186
187 double s_star = std::max(std::sqrt(alpha_star*alpha_star + lambda_star*lambda_star), 0.01);
188
189 slip_angle = s_star * pac_param_.kAlphaP;
190 slip_ratio = s_star * pac_param_.kLambdaP;
191
192 double fy_pure = tire_load * pac_param_.Dlat * std::sin(pac_param_.Clat * std::atan(pac_param_.Blat * slip_angle));
193 double fx_pure = tire_load * pac_param_.Dlon * std::sin(pac_param_.Clon * std::atan(pac_param_.Blon * slip_ratio));
194
195 Tire_force tire_force;
196 tire_force.fy = fy_pure * alpha_star / s_star;
197 tire_force.fx = fx_pure * lambda_star / s_star;
198
199 return tire_force;
200}

202void VehicleDynamics::kinematic_correction(){
203 double lambda = 1 - std::clamp((vx_ - 2.0)/(5.0 - 2.0),0.0,1.0);
204 double r_kinematic = std::tan(delta_) * vx_ / kWheelBase;
205 double vy_kinematic = r_kinematic * kLr;

216void VehicleDynamics::update_torque_cmd(){
217 double total_fx_cmd = input_acc_ * kMass;
218
219 if(kTorqueVectoring){
220 double target_r = vx_ * std::tan(delta_) / kWheelBase;
221
222 double target_mz = kTVKp * (target_r - r_);
223
224 double fz_total = kMass * kG + 0.5 * kAirDensity * kCLA * vx_*vx_;
225 double fz_front_mean = 0.5*(tire_loads_.fl_ + tire_loads_.fr_);
226 double fz_rear_mean = 0.5*(tire_loads_.rl_ + tire_loads_.rr_);
227
228 torque_cmd_.fl_ = kTireDynRadius / fz_total * (fz_front_mean * total_fx_cmd - tire_loads_.fl_ * 2*target_mz/kTrackWidth);
229 torque_cmd_.fr_ = kTireDynRadius / fz_total * (fz_front_mean * total_fx_cmd + tire_loads_.fr_ * 2*target_mz/kTrackWidth);
230 torque_cmd_.rl_ = kTireDynRadius / fz_total * (fz_rear_mean * total_fx_cmd - tire_loads_.rl_ * 2*target_mz/kTrackWidth);
231 torque_cmd_.rr_ = kTireDynRadius / fz_total * (fz_rear_mean * total_fx_cmd + tire_loads_.rr_ * 2*target_mz/kTrackWidth);
232 }else{
233 torque_cmd_.fl_ = 0.2 * total_fx_cmd * kTireDynRadius;
234 torque_cmd_.fr_ = 0.2 * total_fx_cmd * kTireDynRadius;
235 torque_cmd_.rl_ = 0.3 * total_fx_cmd * kTireDynRadius;
236 torque_cmd_.rr_ = 0.3 * total_fx_cmd * kTireDynRadius;
237 }
238
239 torque_cmd_.fl_ = std::clamp(torque_cmd_.fl_, kTorqueMin, kTorqueMax);
240 torque_cmd_.fr_ = std::clamp(torque_cmd_.fr_, kTorqueMin, kTorqueMax);
241 torque_cmd_.rl_ = std::clamp(torque_cmd_.rl_, kTorqueMin, kTorqueMax);
242 torque_cmd_.rr_ = std::clamp(torque_cmd_.rr_, kTorqueMin, kTorqueMax);
243
244}
